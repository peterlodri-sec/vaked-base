"""ralphcore — pure logic for the ralph decision loop (no I/O)."""
from __future__ import annotations
import json
import os
from dataclasses import dataclass


@dataclass(frozen=True)
class Repo:
    name: str
    path: str   # absolute, expanded
    gh: str     # owner/repo


def load_repos(config_path: str) -> list[Repo]:
    with open(config_path, encoding="utf-8") as fh:
        cfg = json.load(fh)
    out = []
    for r in cfg["repos"]:
        out.append(Repo(name=r["name"],
                        path=os.path.abspath(os.path.expanduser(r["path"])),
                        gh=r["gh"]))
    return out
```

def select_candidate(candidates: list[dict]) -> "dict | None":
    """Highest-urgency unaddressed candidate (ties → first in list); if all are
    addressed, the highest-urgency one (to advance with new info). None if empty."""
    if not candidates:
        return None
    def key(c):
        return int(c.get("urgency", 0) or 0)
    unaddressed = [c for c in candidates if not c.get("addressed", False)]
    pool = unaddressed if unaddressed else candidates
    best = pool[0]
    for c in pool[1:]:
        if key(c) > key(best):
            best = c
    return best
```

def next_repo(names: list[str], current: "str | None",
              unavailable: set) -> "str | None":
    """The next available repo after ``current`` (round-robin, wrapping). On
    first run (current=None) the first available. None if all unavailable."""
    available = [n for n in names if n not in unavailable]
    if not available:
        return None
    if current is None or current not in names:
        return available[0]
    start = names.index(current)
    n = len(names)
    for i in range(1, n + 1):
        cand = names[(start + i) % n]
        if cand in available:
            return cand
    return available[0]
```

def _bar(frac: float, width: int = 20) -> str:
    frac = max(0.0, min(1.0, frac))
    fill = int(round(frac * width))
    return "[" + "#" * fill + "-" * (width - fill) + "]"


def render_dashboard(status: "dict | None", now_epoch: int,
                     last_step_epoch: int) -> str:
    if not status:
        return "ralph — no supervisor running (status.json missing or empty)\n"
    lines = []
    dot = "●" if status.get("running") else "○"
    since = max(0, int(now_epoch - last_step_epoch))
    lines.append("ralph eye  %s  current=%s  iter=%d  (%ds since last step)" % (
        dot, status.get("current", "-"), status.get("iteration", 0), since))
    total = float(status.get("total_cost", 0.0))
    budget = float(status.get("budget_total", 0.0)) or 1e-9
    lines.append("spend  $%.4f / $%.2f  %s" % (total, budget, _bar(total / budget)))
    lines.append("")
    lines.append("%-18s %6s  %-40s %8s" % ("repo", "n", "last decision", "cost$"))
    for name, r in sorted(status.get("repos", {}).items()):
        lines.append("%-18s %6d  %-40.40s %8.4f" % (
            name, r.get("entries", 0), r.get("last_title", "-"), r.get("cost", 0.0)))
    recent = status.get("recent", [])
    if recent:
        lines.append("")
        lines.append("recent decisions:")
        for r in recent[:5]:
            lines.append("  [%s] %s — %s" % (r.get("repo", "?"), r.get("date", ""),
                                             r.get("title", "")))
    return "\n".join(lines) + "\n"
```

import ralphcore as C  # noqa: E402

REPO_HOME = os.path.abspath(os.path.join(HERE, "..", ".."))   # vaked-base
DECISIONS_DIR = os.path.join(REPO_HOME, "docs", "decisions")
STATE_DIR = os.path.join(HERE, "state")
STATUS_PATH = os.path.join(STATE_DIR, "status.json")
OPENROUTER_URL = "https://openrouter.ai/api/v1/chat/completions"
DEFAULT_S1 = "qwen/qwen3-235b-a22b-thinking-2507"
DEFAULT_S2 = "deepseek/deepseek-v4-flash"


def _run(cmd: list[str], cwd: str, timeout: int = 30) -> str:
    """Run a read-only command, return stdout (empty string on any failure)."""
    try:
        r = subprocess.run(cmd, cwd=cwd, capture_output=True, text=True,
                           timeout=timeout)
        return r.stdout if r.returncode == 0 else ""
    except Exception:
        return ""


def gather_context(repo: C.Repo, git_log_window: int, compact: bool) -> str:
    """Read-only project state for a repo: issues, recent commits, key docs.
    compact=True trims for stage 1; full text for stage 2."""
    parts = []
    issues = _run(["gh", "issue", "list", "--repo", repo.gh, "--state", "open",
                   "--limit", "40", "--json", "number,title,body"], repo.path)
    if issues:
        data = json.loads(issues)
        if compact:
            parts.append("## Open issues\n" + "\n".join(
                "#%d %s" % (i["number"], i["title"]) for i in data))
        else:
            parts.append("## Open issues\n" + "\n".join(
                "### #%d %s\n%s" % (i["number"], i["title"],
                                    (i.get("body") or "")[:4000]) for i in data))
    else:
        parts.append("## Open issues\n(unavailable)")
    gitlog = _run(["git", "log", "--oneline", "-n", str(git_log_window)], repo.path)
    parts.append("## Recent commits\n" + (gitlog or "(unavailable)"))
    for rel in ("README.md", "CLAUDE.md", "AGENTS.md"):
        fp = os.path.join(repo.path, rel)
        if os.path.exists(fp):
            txt = open(fp, encoding="utf-8", errors="replace").read()
            parts.append("## %s\n%s" % (rel, txt if not compact else txt[:1500]))
    return "\n\n".join(parts)


def openrouter_call(model: str, messages: list[dict], *, api_key: str,
                    temperature: float, max_tokens: int,
                    reasoning: "dict | None" = None,
                    response_format: "dict | None" = None,
                    seed: "int | None" = None, retries: int = 3) -> dict:
    """POST to OpenRouter chat-completions; return the parsed JSON. Bounded
    retry with backoff on non-200 / network error."""
    body = {"model": model, "messages": messages,
            "temperature": temperature, "top_p": 0.95, "max_tokens": max_tokens}
    if reasoning is not None:
        body["reasoning"] = reasoning
    if response_format is not None:
        body["response_format"] = response_format
    if seed is not None:
        body["seed"] = seed
    data = json.dumps(body).encode("utf-8")
    last = None
    for attempt in range(retries):
        req = urllib.request.Request(OPENROUTER_URL, data=data, method="POST",
            headers={"Authorization": "Bearer " + api_key,
                     "Content-Type": "application/json"})
        try:
            with urllib.request.urlopen(req, timeout=120) as resp:
                return json.loads(resp.read().decode("utf-8"))
        except Exception as e:   # noqa: BLE001 — bounded retry then re-raise
            last = e
            time.sleep(2 ** attempt)
    raise RuntimeError("OpenRouter call failed after %d tries: %s" % (retries, last))

def cmd_run(args) -> int:
    repos = C.load_repos(args.repos)
    names = [r.name for r in repos]
    by_name = {r.name: r for r in repos}
    status = read_status() or {
        "running": True, "current": None, "iteration": 0,
        "total_cost": 0.0, "budget_total": args.budget_total,
        "repos": {n: {"entries": 0, "last_title": "-", "cost": 0.0} for n in names},
        "recent": [], "last_step_epoch": 0,
    }
    status["running"] = True
    status["budget_total"] = args.budget_total
    iters = 0
    try:
        while True:
            if status["total_cost"] >= args.budget_total:
                print("budget cap reached ($%.2f) — stopping" % args.budget_total)
                break
            if args.max_iters and iters >= args.max_iters:
                print("max-iters reached — stopping"); break
            unavailable = {r.name for r in repos if not os.path.isdir(r.path)}
            nxt = C.next_repo(names, status["current"], unavailable)
            if nxt is None:
                print("no available repos — stopping", file=sys.stderr); break
            status["current"] = nxt
            status["iteration"] += 1
            iters += 1
            _supervised_decide(args, by_name[nxt], status)   # updates status in place
            status["last_step_epoch"] = int(time.time())
            write_status(status)
            if args.max_iters and iters >= args.max_iters:
                continue
            time.sleep(args.interval)
    except KeyboardInterrupt:
        print("\nSIGINT — shutting down")
    finally:
        status["running"] = False
        write_status(status)
    return 0

def _supervised_decide(args, repo, status) -> None:
    """Run one decide iteration for ``repo``; fold cost + result into ``status``.
    Any failure is contained (logged) so the supervisor keeps running."""
    api_key = os.environ.get("OPENROUTER_API_KEY")
    if not api_key:
        print("OPENROUTER_API_KEY not set — cannot decide", file=sys.stderr)
        return
    try:
        compact = gather_context(repo, args.git_log_window, compact=True)
        s1_msgs = C.build_stage1_messages(repo.name, compact, _prior_titles(repo.name))
        before = len(_prior_titles(repo.name))
        # reuse the live path but capture cost; _decide_live prints + appends.
        rc = _decide_live(args, repo, s1_msgs, api_key)
        after = _prior_titles(repo.name)
        rep = status["repos"].setdefault(repo.name,
                                         {"entries": 0, "last_title": "-", "cost": 0.0})
        if len(after) > before:
            rep["entries"] = len(after)
            rep["last_title"] = after[-1]
            status["recent"].insert(0, {"repo": repo.name, "date": _today(),
                                        "title": after[-1]})
            status["recent"] = status["recent"][:10]
    except Exception as e:   # noqa: BLE001 — contain per-iteration failures
        print("iteration failed for %s: %s" % (repo.name, e), file=sys.stderr)
